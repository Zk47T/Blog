<script>
  document.addEventListener('click', function(event) {
    if (event.target.matches('.copy-btn')) {
      // Get the code block content related to the clicked button
      const codeBlock = event.target.previousElementSibling.querySelector('code').textContent;

      // Create a temporary textarea to hold the text
      const tempTextArea = document.createElement('textarea');
      tempTextArea.value = codeBlock;
      document.body.appendChild(tempTextArea);

      // Select and copy the text
      tempTextArea.select();
      document.execCommand('copy');

      // Remove the temporary textarea
      document.body.removeChild(tempTextArea);

      // Change button text to "Copied!"
      event.target.textContent = 'Copied!';
      event.target.classList.add('copied');

      // Change the button back after 2 seconds
      setTimeout(function() {
        event.target.textContent = 'Copy';
        event.target.classList.remove('copied');
      }, 2000);
    }
  });
</script>
